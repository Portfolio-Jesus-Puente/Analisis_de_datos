# 1. Crear columna con da en ingls
df["weekday"] = df["day"].dt.day_name()

# 2. Traducir al espaol
df["weekday"] = df["weekday"].replace({
    "Monday": "Lunes",
    "Tuesday": "Martes",
    "Wednesday": "Mircoles",
    "Thursday": "Jueves",
    "Friday": "Viernes",
    "Saturday": "Sbado",
    "Sunday": "Domingo"
})

df["week"] = df["day"].dt.isocalendar().week   #  ESTA ES LA QUE FALTABA


# 3. Orden correcto en espaol
orden = ["Lunes", "Martes", "Mircoles", "Jueves", "Viernes", "Sbado", "Domingo"]

# 4. Crear tabla dinmica usando los das ya traducidos
pivot = df.pivot_table(values="energy_sum", index="week", columns="weekday", aggfunc="mean")

# 5. Reordenar columnas
pivot = pivot[orden]

# 6. Dibujar heatmap
sns.heatmap(pivot, cmap="YlOrRd")
plt.title("Mapa de calor del consumo por semana y da")
plt.ylabel("Semana")       #  Eje Y en espaol
plt.xlabel(" ")
plt.show()
